### №4.93

$$
\sin \left(  \frac{z}{z+1}  \right)
$$
$z_{0}=-1; \ z_{1}=\infty$

$$
z=-1: \lim_{ z \to -1 } \sin \left(  \frac{z}{z+1}  \right)= \lim_{ z \to -1} \sin \left(  -\frac{1}{0}  \right)\implies\not\exists \lim \text{существенная точка}
$$
$$
z=\infty:\lim_{ z \to \infty } \sin \left( \frac{z}{z+1} \right)=\lim_{ z \to \infty } \sin \left( 1 - \frac{1}{z+1} \right)=C\implies z=\infty\text{ устранимая}
$$
$$
\sin\left(  \frac{z}{z+1}  \right)=\sin\left( \frac{t-1}{t} \right)=\sin\left( 1- \frac{1}{t} \right)=\sin(1)\cos\left( \frac{1}{t} \right)-\sin\left( \frac{1}{t} \right)\cos(1)=
$$
$$
\sin\left( \frac{z}{z+1} \right)=\sin(1)\sum_{n=-\infty}^{+\infty}(-1)^{n-1} \frac{t^{2n-2}}{(2n-2)!}=\cos(1)\sum_{n=-\infty}^{+\infty}(-1)^{n-1} \left( \frac{t^{-(2n-1)}}{(2n-1)!} \right)
$$
$$
t=z+1, z=-1: \sin(1)\cdot \left( 1- \frac{1}{2!\cdot t^{2}}+ \frac{1}{t^{4}} \right) -\cos(1)\left(  \frac{1}{t} - \frac{1}{3!\cdot t^{3}} \right)
$$
$$
\mathrm{Res}_{z=-1}=-\cos(1); \text{Res}_{z=\infty}=\cos(1).
$$

### №4.79

$$
f(z)= \frac{1}{z^{3}-z^{5}}= \frac{1}{z^{3}(1-z^{2})}= \frac{1}{z^{3}(1-z)(1+z)}
$$
$z_{0}=-1; \ z_{1}=0; \ z_{2}=1\ z_{3}=\infty$

1. 
$$
\lim_{ z \to -1 } \frac{1}{z^{3}-z^{5}}=\infty\implies\text{полюс 1-ого порядка}
$$
2. 
$$
\lim_{ z \to z_{1} } \frac{1}{z^{3}-z^{5}}=\infty\implies\text{полюс 3-ого}
$$
3. 
$$
\lim_{ z \to z_{2} } \frac{1}{z^{3}-z^{5}}=\infty\implies\text{полюс 1-го}
$$
 4. 
$$
\lim_{ z \to 1 } \frac{1}{z^{3}-z^{5}}=0\implies\text{устранимая}
$$

$$
z=-1: \frac{1}{z^{3}(1-z^{2})}= \frac{1}{z^{3}} \sum_{n=-\infty}^{+\infty} z^{2n}=\sum_{n=0}^{+\infty}z^{2n-3}
$$
$$
\mathrm{Res}_{z_{1}=0} f(z)=1
$$
$z_{0}$ - полюс $n$-го порядка, то 
$$
\text{Res}f(z_{0})=\frac{1}{(n-1)!}\lim_{ z \to z_{0} } \frac{d^{n-1}(z-z_{0})^{n}}{dz^{n-1}}\cdot f(z)
$$
Рассмотрим $z_{2}=1$
$$
\text{Res}f(z=\pm1)=\lim_{ z \to z_{2} } (z\pm1)^{1} \frac{1}{z^{3}-z^{5}}=\lim_{ z \to 1 }  \frac{z-1}{(1-z)(1+z)z^{3}}=\lim_{ z \to-1 }  -\frac{1}{(1+z)z^{3}}=-\frac{1}{2}
$$
$$
\text{Res}f(z=\infty)=0
$$

### №4.81

$$
f(z)= \frac{z^{2n}}{(1+z)^{n}}
$$
$z_{1}=-1$ - полюс $n$-го порядка
$z_{2}=\infty$

$$
\lim_{ z \to \infty } \frac{z^{2n}}{(1+z)^{n}}=\lim_{ z \to \infty } \frac{1}{\left( \frac{1}{z^{2}}+\frac{1}{z} \right)^{n}}=\infty -\text{полюс }n-ого
$$
$$
\text{Res}_{z=-1}f(z)= \frac{1}{(n-1)!}\cdot \lim_{ z \to -1 } \frac{d^{n-1}}{dz^{n-1}}(z+1)^{n}f(z)=\lim_{ z \to-1 }  \frac{2n!}{(n+1)!(n-1)!}z^{n+1}
$$

$$
\text{Res}_{z=\infty}(f(z))= \frac{2n!}{(n-1)!(n+1)!}(-1)^{n}
$$

### №

$$
\frac{1}{2\pi i} \oint_{C} \sin \frac{1}{z}, \ C\text{ - окр}. |z|=R
$$

$z=0:$ существенная особая точка

$$
\sin \frac{1}{z}= \frac{\sum_{n=0}^{\infty}(-1)^{n}}{(2n+1)!} \left( \frac{1}{z} \right)^{2n+1}= \frac{1}{z} - \frac{1}{3!z^{3}}+ \frac{1}{5!z^{5}}-\dots\to \text{Res}_{z=0}f(z)=1
$$

$$
\frac{1}{2\pi i}\oint_{C}\sin \frac{1}{z}\,dz= \frac{1}{2\pi i}  \cdot 2\pi i \cdot 1 = 1
$$
