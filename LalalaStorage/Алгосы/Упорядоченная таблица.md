**Определение:** *Упорядоченная таблица* - таблица, используемая бинарный поиск вместо линейного поиска.

Из-за особенности бинарного поиска при реализации таблицы мы не сможем использовать списки.

**Операции:**
- Вставка - $O(\log_{2} n)$
- Удаление - $O(\log_{2}n)$
- Поиск - $O(\log_{2}n)$

## Поиск

![[Pasted image 20250609191137.png]]

```c
int search(Table t, int t){
	if (t.csize == 0){
		return err;
	}
	int i = 0;
	int m = n - 1;
	while (i <= m){
		j = (i + m) / 2;
		if (t.data[j].key == key){
			return j;
		}
		if (t.data[j].key > key){
			m = j - 1;
		} else {
			i = j + 1;
		}
	}
	return -1;
}
```

## Вставка

```c
int insert(Table t, int key, Info info){
	if (t.csize == t.msize){
		return err;
	}
	if (search(t, key) != -1){
		return err;
	}
	int i = t.csize;
	while (i >= 0 && t.data[i - 1].key > key){
		t.data[i] = t.data[i-1];
		--i;
	}
	t.data[i].key = key;
	t.data[i].info = info;
	t.csize++;
}
```

## Удаление

```c
int erase(Table t, int key){
	if (t.csize == 0){
		return err;
	}
	int i = search(t, key);
	if (i == -1){
		return -1;
	}
	for (; i + 1 < t.csize; ++i){
		t.data[i] = t.data[i + 1];
	}
	--t.csize;
}
```
